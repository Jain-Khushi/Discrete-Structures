#include<iostream>
using namespace std;
int main()
{
    int v, arr[20][20],p=0, count=0;
    char ed;
    cout<<"\n Enter the number of vertices of the graph:";
    cin>>v;
    cout<<endl;
    for(int i=1;i<=v;i++)
    {
        for(int j=1;j<=v;j++)
        {
            cout<<"Does the graph have edge between ("<<i<<" , "<<j<<") ? Type Y/N. " ;
            cin>>ed;
            if(ed=='Y')
            {
                 arr[i][j]=1; 
            }
            else 
            {
                arr[i][j]=0;    
            }
            
       }
    }
   cout<<"\nAdjacency matrix :\n";
   for(int i=1;i<=v;i++)
     {
        for(int j=1;j<=v;j++)
        {
            cout<<arr[i][j]<<" ";
        }
        cout<<endl;
    }
    for(int i=1;i<=v;i++)
     {
        for(int j=1;j<=v;j++)
        {
          
            if(i!=j&&arr[i][j]==1)
            {
                   ++count;
                 
            }
            else if(i==j&&arr[i][j]!=1)
            {
               ++p;
             
              
            }
           
        }
    }
    if((count==v*v-v)&&p==v)
    {
          cout<<"\nIt is a Complete graph.";
    }
    else
        {
             cout<<"\nIt is an Incomplete graph.";

        }

    return 0;
}
