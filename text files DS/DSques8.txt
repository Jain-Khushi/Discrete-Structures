void BubbleSort(int AR[],int size){
    int tmp,no_of_comparison;
    for(int i=0; i<size; i++){
        no_of_comparison=0;
        for(int j=0; j<(size-1)-i; j++){
            ++no_of_comparison;
            
            if(AR[j]>AR[j+1]){
                tmp = AR[j];
                AR[j] = AR[j+1];
                AR[j+1] = tmp;
            }
        }
        cout<<"\nArray after iteration "<< i+1<<" is : ";
        for(int k=0; k<size; k++)
            cout<<AR[k]<<" ";
        cout<<"\nNumber of comparisons after iteration "<<i+1 <<" is : "<<no_of_comparison;
        cout<<endl;
    }
}
